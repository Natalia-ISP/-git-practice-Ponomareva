git init - создает новый локальный репозиторий.
git remote add origin <URL> - связывает лок. репозиторий с удаленным.
git remote set-url origin <новый_URL> — изменяет URL удалённого
репозитория с псевдонимом origin.

touch имя_файла.txt - создает файл
mkdir имя_папки - создает папку

git add . - добавляет все в индекс для следщ коммита
git commit -m "сообщение" - создает коммита
git push - отправляет все коммиты в репозиторий

git status - состояние гит

cd имя_папки - перейти в другую папку

clear - очистить вывод в терминале
git log - показ истории коммитов

git checkout -b new-имя_ветки - создание новой ветки с переключением
git branch - просмотр веток
git switch -c имя_ветки - создание и переключение
git switch имя_ветки - переключение на сущ ветку
git switch - - переключение на предыдущую ветку
